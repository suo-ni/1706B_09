###xiaoshuo-ss-sfff-fe  变为驼峰xiaoshuoSsSfffFe
function getCamelCase(str) {
    var arr = str.split( '-' );
    return arr.map( function( item, index ) {
        console.log(item)
        if( index === 0 ){
            return item;
        }else{
            return item.charAt(0).toUpperCase() + item.slice( 1 );
        }
    }).join('');
}
    console.log(getCamelCase("xiaoshuo-ss-sfff-fe"))
